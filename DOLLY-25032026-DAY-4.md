class Solution {
public:
    int missingNumber(vector<int>& nums) {
        int maxVal = *max_element(nums.begin(), nums.end());
        int n =nums.size();
        int total=n*(n+1)/2 ;
        int sum=0;
        for(int i=0;i<nums.size();i++){
            sum =sum+nums[i];
        }
        int missingnumber = total-sum ;
        return missingnumber ;
        
    }
};
