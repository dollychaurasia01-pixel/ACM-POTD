class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        ListNode*temp=head;
        ListNode*curr=nullptr;
        ListNode*prev=nullptr;
          
          while(temp!=nullptr){
           curr=temp->next;
            temp->next=prev;
            prev=temp;
            temp=curr;
           
          }
           head=prev;
          return head;
    }
};
